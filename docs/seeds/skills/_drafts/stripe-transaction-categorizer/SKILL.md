# stripe-transaction-categorizer

Automatically assigns an accounting ledger category to an incoming Stripe transaction based on predefined categories.

## Implementation plan

Receives a raw Stripe transaction and a list of accounting categories. Uses an LLM to analyze transaction details (description, type, amount, metadata) and infer the most appropriate category from the provided list, then augments the transaction with the assigned category and a confidence score.

import { Context } from '@day14/skill-sdk';

interface StripeTransaction {
  id: string;
  amount: number;
  currency: string;
  description: string | null;
  type: string; // e.g., 'charge', 'payout', 'refund'
  metadata?: Record<string, any>;
  // ... other relevant Stripe fields
}

interface CategorizedStripeTransaction extends StripeTransaction {
  accountingCategory: string;
  categoryConfidenceScore?: number;
}

export async function run(ctx: Context): Promise<void> {
  const { stripeTransaction, accountingCategories } = ctx.inputs as {
    stripeTransaction: StripeTransaction;
    accountingCategories: string[];
  };

  if (!stripeTransaction || !accountingCategories || accountingCategories.length === 0) {
    throw new Error("Missing stripeTransaction or accountingCategories input.");
  }

  const transactionDetails = JSON.stringify({
    id: stripeTransaction.id,
    amount: stripeTransaction.amount,
    currency: stripeTransaction.currency,
    description: stripeTransaction.description,
    type: stripeTransaction.type,
    metadata: stripeTransaction.metadata,
  }, null, 2);

  const prompt = `Categorize the following Stripe transaction into one of these accounting categories: ${accountingCategories.join(', ')}.
  Provide only the best-fit category name and a confidence score (0-1).

  Stripe Transaction Details:
  ${transactionDetails}

  Output format: {"category": "Category Name", "confidence": 0.X}`;

  // In a real Day14 OS environment, ctx.llm.chat() or similar would be preferred.
  // Using fetch directly for demonstration as per instructions.
  const llmResponse = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`, // Assumes API key is available in env
    },
    body: JSON.stringify({
      model: 'gpt-3.5-turbo-0125', // Or a more capable model like gpt-4-turbo
      messages: [{ role: 'user', content: prompt }],
      response_format: { type: 'json_object' },
      temperature: 0.2 // Keep temperature low for deterministic categorization
    }),
  });

  if (!llmResponse.ok) {
    const errorBody = await llmResponse.text();
    throw new Error(`LLM API error: ${llmResponse.status} - ${errorBody}`);
  }

  const llmResult = await llmResponse.json();
  const parsedOutput = JSON.parse(llmResult.choices[0].message.content);

  let accountingCategory = parsedOutput.category;
  const categoryConfidenceScore = parsedOutput.confidence;

  // Validate if the LLM-suggested category is in the provided list
  if (!accountingCategories.includes(accountingCategory)) {
    ctx.log.warn(`LLM suggested category "${accountingCategory}" not in provided list. Falling back to "Uncategorized".`);
    accountingCategory = "Uncategorized";
  }

  const categorizedTransaction: CategorizedStripeTransaction = {
    ...stripeTransaction,
    accountingCategory,
    categoryConfidenceScore
  };

  ctx.outputs.categorizedTransaction = categorizedTransaction;
  ctx.outputs.categoryConfidenceScore = categoryConfidenceScore;
}
