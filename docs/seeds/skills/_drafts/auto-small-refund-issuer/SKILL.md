# auto-small-refund-issuer

Automatically evaluates an order for refund eligibility and issues refunds under a specified monetary threshold, recording the action in an audit log.

## Implementation plan

Fetch order details from the order management system using the provided orderId. Evaluate refund eligibility based on predefined business rules and check if the refund amount is below maxAutoRefundAmount. If eligible and within the threshold, call the payment gateway API to issue the refund and then use the 'audit-log-generator' skill to record the transaction.

import { Day14Context } from '@day14/skill-sdk';

interface AutoSmallRefundIssuerInputs {
  orderId: string;
  maxAutoRefundAmount?: number; // Default to 50
}

interface AutoSmallRefundIssuerOutputs {
  refundStatus: 'ISSUED' | 'ELIGIBLE_PENDING_MANUAL' | 'NOT_ELIGIBLE' | 'ERROR';
  refundAmountIssued?: number;
  auditLogEntryId?: string;
  reason: string;
}

export async function run(ctx: Day14Context<AutoSmallRefundIssuerInputs, AutoSmallRefundIssuerOutputs>) {
  const { orderId, maxAutoRefundAmount = 50 } = ctx.inputs;

  ctx.log.info(`Processing auto-refund for order ${orderId} with max amount $${maxAutoRefundAmount}`);

  try {
    // Simulate fetching order details from an external OMS/CRM
    const orderResponse = await ctx.fetch(`https://api.example.com/orders/${orderId}`);
    if (!orderResponse.ok) {
      throw new Error(`Failed to fetch order ${orderId}: ${orderResponse.statusText}`);
    }
    const order = await orderResponse.json();

    // Simulate refund eligibility check (e.g., return window, item condition, no prior refunds)
    const isRefundEligible = order.status === 'returned' && order.returnWindowValid && !order.hasPriorRefund;
    const refundAmount = order.totalAmount; // Assuming full refund for simplicity

    if (!isRefundEligible) {
      return ctx.complete({
        refundStatus: 'NOT_ELIGIBLE',
        reason: 'Order not eligible for refund based on current rules.'
      });
    }

    if (refundAmount > maxAutoRefundAmount) {
      return ctx.complete({
        refundStatus: 'ELIGIBLE_PENDING_MANUAL',
        reason: `Refund amount ($${refundAmount}) exceeds auto-refund threshold ($${maxAutoRefundAmount}).`
      });
    }

    // Simulate issuing refund via payment gateway API
    const refundResponse = await ctx.fetch('https://api.paymentgateway.com/refunds', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${ctx.secrets.get('PAYMENT_GATEWAY_API_KEY')}` },
      body: JSON.stringify({ orderId, amount: refundAmount })
    });

    if (!refundResponse.ok) {
      throw new Error(`Failed to issue refund for order ${orderId}: ${refundResponse.statusText}`);
    }

    // Call the existing 'audit-log-generator' skill to record the action
    const auditLogResult = await ctx.skills.call('audit-log-generator', {
      action: 'auto-refund-issued',
      entityType: 'order',
      entityId: orderId,
      details: { amount: refundAmount, initiatedBy: 'auto-small-refund-issuer' }
    });

    return ctx.complete({
      refundStatus: 'ISSUED',
      refundAmountIssued: refundAmount,
      auditLogEntryId: auditLogResult.logEntryId as string,
      reason: `Refund of $${refundAmount} successfully issued.`
    });

  } catch (error: any) {
    ctx.log.error(`Error processing auto-refund for order ${orderId}: ${error.message}`);
    return ctx.complete({
      refundStatus: 'ERROR',
      reason: `An error occurred during refund processing: ${error.message}`
    });
  }
}
