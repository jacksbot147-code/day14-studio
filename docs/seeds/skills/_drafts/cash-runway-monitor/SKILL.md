# cash-runway-monitor

Monitors daily cash balances from Stripe and bank accounts, calculates the projected runway, and alerts if it falls below a specified threshold.

## Implementation plan

The skill will use the provided Stripe API key to fetch the current Stripe balance. It will then use the bank API endpoint and credentials to fetch the primary bank account balance. These balances will be summed to determine the total cash position. Using the provided daily burn rate, the skill will calculate the projected runway in days and compare it against the threshold, generating an alert if necessary.

```typescript
import { Context } from '@day14/jack';

interface CashRunwayMonitorConfig {
  bankApiEndpoint: string;
  dailyBurnRateUsd: number;
  runwayThresholdDays: number;
}

interface CashRunwayMonitorSecrets {
  stripeApiKey: string;
  bankApiCredentials: { [key: string]: string };
}

export async function run(ctx: Context<CashRunwayMonitorConfig, CashRunwayMonitorSecrets>) {
  const { config, getSecret, log } = ctx;
  const { bankApiEndpoint, dailyBurnRateUsd, runwayThresholdDays } = config;

  const stripeApiKey = await getSecret('stripeApiKey');
  const bankApiCredentials = await getSecret('bankApiCredentials');

  let stripeBalanceUsd = 0;
  try {
    const stripeResponse = await ctx.fetch('https://api.stripe.com/v1/balance', {
      headers: {
        'Authorization': `Bearer ${stripeApiKey}`,
        'Content-Type': 'application/json',
      },
    });
    const stripeData = await stripeResponse.json();
    // Assuming Stripe balance is in cents and 'available' is the relevant part
    stripeBalanceUsd = (stripeData.available?.[0]?.amount || 0) / 100;
  } catch (error) {
    log.error('Failed to fetch Stripe balance:', error);
  }

  let bankBalanceUsd = 0;
  try {
    const bankResponse = await ctx.fetch(bankApiEndpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(bankApiCredentials),
    });
    const bankData = await bankResponse.json();
    // Assuming bank API returns a 'balance' field
    bankBalanceUsd = bankData.balance || 0;
  } catch (error) {
    log.error('Failed to fetch bank balance:', error);
  }

  const totalCashBalanceUsd = stripeBalanceUsd + bankBalanceUsd;
  const calculatedRunwayDays = dailyBurnRateUsd > 0 ? totalCashBalanceUsd / dailyBurnRateUsd : Infinity;
  const isRunwayBelowThreshold = calculatedRunwayDays < runwayThresholdDays;

  let alertMessage = '';
  if (isRunwayBelowThreshold) {
    alertMessage = `🚨 WARNING: Cash runway is ${calculatedRunwayDays.toFixed(0)} days, which is below the ${runwayThresholdDays} day threshold! Total cash: $${totalCashBalanceUsd.toFixed(2)}.`;
    log.warn(alertMessage);
  } else {
    log.info(`Cash runway is ${calculatedRunwayDays.toFixed(0)} days. Total cash: $${totalCashBalanceUsd.toFixed(2)}.`);
  }

  return {
    totalCashBalanceUsd,
    calculatedRunwayDays,
    isRunwayBelowThreshold,
    alertMessage
  };
}
```
